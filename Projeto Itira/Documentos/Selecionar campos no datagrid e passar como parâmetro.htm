Selecionar campos no datagrid e passar como parâmetro

//Como pegar os valores dos campos selecionados no datagrid e jogar na variavel de sessão no Delphi 2005/2006.Segue abaixo como ficaria.

// aqui eu pego o registro selecionado
procedure TWebForm1.DataGrid1_ItemCommand(source: System.Object; e: System.Web.UI.WebControls.DataGridCommandEventArgs);
var
  IdAluno: string;
begin
  if (e.CommandName = 'Select') then
  begin
    IdAluno := DataGrid1.DataKeys[e.Item.ItemIndex].ToString();
    AbrirRegistro(IdAluno);
  end;
end;

// aqui eu jogo na session os dados selecionanados no datagrid,
procedure TWebForm1.AbrirRegistro(IdAluno: string);
var
  dr: SqlDataReader;
 
begin
  SqlConnection1.Open();
  try
    SqlCommand2.Parameters['@PID_ALUNO'].Value := IdAluno;
    dr := SqlCommand2.ExecuteReader();
    dr.Read();
    session['id']:= dr['ID_ALUNO'].ToString();
    session['nome']:= dr['NOME_ALUNO'] as string;
    session['email']:= dr['EMAIL'] as string;
    dlCURSO.SelectedValue := dr['ID_CURSO'].ToString();
    dr.Close();
  finally
    SqlConnection1.Close();
  end;
end;

Esta linha de codigo deve ser usada para saber qual é o id_aluno que foi selecionado no datagrid :
DataGrid1.DataKeyField := 'ID_ALUNO';

procedure TWebForm1.AbrirConsulta;
var
  dr: SqlDataReader;
begin
// pega conexão do pool de conexões
  SqlConnection1.Open();
  try
    dr := SqlCommand1.ExecuteReader();
    DataGrid1.DataSource := dr;
    // para saber qual é o id_aluno que foi selecionado
    DataGrid1.DataKeyField := 'ID_ALUNO';
    DataGrid1.DataBind();
  finally
    // devole a conexão para o pool
    SqlConnection1.Close();
  end;

end;
