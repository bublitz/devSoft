Calculando o IMPOSTO DE RENDA
Script
Autor rfabh 04/03/2006

Índice

» IMPOSTO DE RENDA
» Script

CREATE TABLE "IRRF"
(
"COD_IRRF" INTEGER NOT NULL,
"DT_INI" TIMESTAMP NOT NULL,
"FAIXA1" NUMERIC(15, 2),
"TAXA1" NUMERIC(15, 2),
"DEDUCAO1" NUMERIC(15, 2),
"FAIXA2" NUMERIC(15, 2),
"TAXA2" NUMERIC(15, 2),
"DEDUCAO2" NUMERIC(15, 2),
"FAIXA3" NUMERIC(15, 2),
"TAXA3" NUMERIC(15, 2),
"DEDUCAO3" NUMERIC(15, 2),
PRIMARY KEY ("COD_IRRF"
);


CREATE GENERATOR "GEN_IRRF";

SET TERM ^ ;

CREATE PROCEDURE "CALCULA_IRRF"
(
"VALOR" FLOAT,
"DATA" TIMESTAMP
)
RETURNS
(
"IMPOSTO" NUMERIC(15, 4)
)
AS
DECLARE VARIABLE FX1 FLOAT;
DECLARE VARIABLE FX2 FLOAT;
DECLARE VARIABLE FX3 FLOAT;
DECLARE VARIABLE TX1 FLOAT;
DECLARE VARIABLE TX2 FLOAT;
DECLARE VARIABLE TX3 FLOAT;
DECLARE VARIABLE DE1 FLOAT;
DECLARE VARIABLE DE2 FLOAT;
DECLARE VARIABLE DE3 FLOAT;
DECLARE VARIABLE COD_IRRF INTEGER;
BEGIN
/* DEFINIR INICIALMENTE O VALOR COMO 0 */
IMPOSTO = 0;
/* PEGAR A TABELA VALIDA PARA DATA */
SELECT MAX(COD_IRRF) FROM IRRF WHERE DT_INI <= ATA INTO :COD_IRRF;
SELECT MAX(FAIXA1), MAX(FAIXA2), MAX(FAIXA3), MAX(TAXA1),
MAX(TAXA2), MAX(TAXA3), MAX(DEDUCAO1), MAX(DEDUCAO2), MAX(DEDUCAO3)
FROM IRRF WHERE COD_IRRF=:COD_IRRF
INTO :FX1, :FX2, :FX3, :TX1, :TX2, :TX3, E1, E2, E3;
/* EVITANDO VALORES NULOS */
IF (FX1 IS NULL) THEN FX1 = 99999999999;
IF (FX2 IS NULL) THEN FX2 = 99999999999;
IF (FX3 IS NULL) THEN FX3 = 99999999999;
/* EVITANDO VALORES NULOS */
IF (TX1 IS NULL) THEN TX1 = 0;
IF (TX2 IS NULL) THEN TX2 = 0;
IF (TX3 IS NULL) THEN TX3 = 0;
/* EVITANDO VALORES NULOS */
IF (DE1 IS NULL) THEN DE1 = 0;
IF (DE2 IS NULL) THEN DE2 = 0;
IF (DE3 IS NULL) THEN DE3 = 0;
/* EXECUTAR OS CALCULOS */
IF (VALOR > FX1) THEN IMPOSTO = VALOR * (TX1 / 100) - DE1;
IF (VALOR > FX2) THEN IMPOSTO = VALOR * (TX2 / 100) - DE3;
IF (VALOR > FX3) THEN IMPOSTO = VALOR * (TX3 / 100) - DE3;
/* VERIFICAR ALGUM ERRO */
IF (IMPOSTO IS NULL) THEN IMPOSTO = 0;
IF (IMPOSTO < 0) THEN IMPOSTO = 0;
/* RETORNAR OS VALORES */
FOR SELECT 0 AS ERRO FROM RDB$DATABASE INTO :TX1 DO SUSPEND;
END
^
CREATE TRIGGER "IST_IRRF" FOR "IRRF"
ACTIVE BEFORE INSERT POSITION 0
as
begin
new.cod_irrf=gen_id(gen_irrf, 1);
IF (NEW.FAIXA1 IS NULL) THEN NEW.FAIXA1 = 99999999999;
IF (NEW.FAIXA2 IS NULL) THEN NEW.FAIXA2 = 99999999999;
IF (NEW.FAIXA3 IS NULL) THEN NEW.FAIXA3 = 99999999999;
end
^
COMMIT WORK ^
SET TERM ;^
IMPOSTO DE RENDA
